#include <LiquidCrystal.h>

int Pir = 9;
int Led = 13;
int ButtonValue9;
int pirState = LOW;         
int val = 0;
int i = 0;
LiquidCrystal lcd(12, 11, 5, 4, 3, 2);
  
void setup()
{
  pinMode(Led, OUTPUT); 
  pinMode(Pir, INPUT);
  lcd.begin(16, 2);
//  lcd.print("DETECT "+String(pirState));
  lcd.print("Initializing");
  while(i<=100){
  		lcd.setCursor(0,1);
		  lcd.print(i);
    	lcd.setCursor(3,1);
    	lcd.print("%");
    	i++;
  	}
    delay(2000);
    lcd.clear();
}

void loop()
{
  lcd.setCursor(0, 1);
  lcd.print(millis() / 1000);
  
  ButtonValue9 = digitalRead(Pir); 
  
  val = digitalRead(Pir); 
  if (val == HIGH)
  {
    digitalWrite(Led, HIGH); 
    lcd.print("DETECT "+String(pirState));
   
    if (pirState == LOW)
    {
      
      pirState = HIGH;
    }
  } 
  	else
  
    digitalWrite(Led, LOW);
    
    if (pirState == HIGH)
    {
      lcd.print("NO DETECT "+String(pirState));
      pirState = LOW;
    }
  
  if(ButtonValue9 == HIGH) 
  {
    digitalWrite(Led,HIGH); 
  }
  else
  {
    digitalWrite(Led,LOW); 
  }
}
